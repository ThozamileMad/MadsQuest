<script>
    // GAME STATE
    let stats = { life: 100, mana: 50, gold: 200 };
    let history = []; // For the "Go Back" feature
    let checkpoint = "You wake up in the ruins...";

    function updateUI() {
        document.getElementById('lifeVal').innerText = stats.life;
        document.getElementById('manaVal').innerText = stats.mana;
        document.getElementById('goldVal').innerText = stats.gold;
    }

    function makeChoice(text, id) {
        // Save current state to history before changing
        history.push({text: document.getElementById('storyText').innerText, stats: {...stats}});
        
        document.getElementById('storyText').innerText = text;
        updateUI();
    }

    // LUCK BOOST LOGIC
    function tryLuck() {
        if (stats.gold >= 50) {
            stats.gold -= 50;
            const roll = Math.random();
            if (roll > 0.5) {
                const gain = Math.floor(Math.random() * 20) + 5;
                stats.life += gain;
                document.getElementById('luckResult').innerText = `Success! Gained ${gain} Life!`;
            } else {
                const gain = Math.floor(Math.random() * 15) + 5;
                stats.mana += gain;
                document.getElementById('luckResult').innerText = `Magic flows! Gained ${gain} Mana!`;
            }
            updateUI();
        } else {
            document.getElementById('luckResult').innerText = "Not enough gold!";
        }
    }

    // NAVIGATION LOGIC
    function goBack(type) {
        if (type === 'redo' && history.length > 0) {
            let last = history.pop();
            stats = last.stats;
            document.getElementById('storyText').innerText = last.text;
        } else if (type === 'restart') {
            location.reload();
        } else if (type === 'checkpoint') {
            document.getElementById('storyText').innerText = checkpoint;
        }
        updateUI();
        closeModal('navModal');
    }

    // SETTINGS LOGIC
    function setTheme(t) { document.documentElement.setAttribute('data-theme', t); }
    function setSize(s) { document.documentElement.setAttribute('data-size', s); }

    // MODAL HELPERS
    function openModal(id) { document.getElementById(id).style.display = 'flex'; }
    function closeModal(id) { document.getElementById(id).style.display = 'none'; }

    window.onclick = function(event) {
        if (event.target.className === 'modal') {
            event.target.style.display = "none";
        }
    }
</script>
